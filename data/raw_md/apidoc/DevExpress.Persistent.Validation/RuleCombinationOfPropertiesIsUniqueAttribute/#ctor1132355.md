    type: DevExpress.Persistent.Validation.DefaultContexts

    type: System.String
    description: A string specifying the properties whose combination must be unique. The property names can be separated by the comma, semicolon and whitespace characters.
  - id: messageTemplate
    type: System.String
    description: A string value representing information defining the current rule. This value is set for the [RuleBaseAttribute.CustomMessageTemplate](xref:DevExpress.Persistent.Validation.RuleBaseAttribute.CustomMessageTemplate) property.
