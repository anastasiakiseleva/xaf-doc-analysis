syntax:
  content: public RuleRangeAttribute(string id, string targetContextIDs, object minimumValue, object maximumValue)
  parameters:
  - id: id
    type: System.String
    description: A string value representing the current rule identifier. You can pass an empty value as the _id_ parameter. An appropriate identifier will begenerated automatically, and set for the **ID** property of the **Validation** | **Rules** | **Rule** node in the [Application Model](xref:112580).
  - id: targetContextIDs
    type: System.String
    description: A string value that specifies a list of identifiers of the contexts when the current rule should be checked. This value is set for the [RuleBaseAttribute.TargetContextIDs](xref:DevExpress.Persistent.Validation.RuleBaseAttribute.TargetContextIDs) property.
  - id: minimumValue
    type: System.Object
    description: An object that represents the left end point of the range that should contain the target property's value.
  - id: maximumValue
    type: System.Object
    description: An object that represents the right end point of the range that contains the target property's value.
