syntax:
  content: public RuleIsReferencedAttribute(string id, string targetContextIDs, Type looksFor, string referencePropertyName, string messageTemplate)
  parameters:
  - id: id
    type: System.String
    description: A string value representing the current rule identifier. You can pass an empty value as the _id_ parameter. An appropriate identifier will begenerated automatically, and set for the **ID** property of the **Validation** | **Rules** | **Rule** node in the [Application Model](xref:112580).
  - id: targetContextIDs
    type: System.String
    description: A string value that specifies a list of identifiers of the contexts when the current rule should be checked. This value is set for the [RuleBaseAttribute.TargetContextIDs](xref:DevExpress.Persistent.Validation.RuleBaseAttribute.TargetContextIDs) property.
  - id: looksFor
    type: System.Type
    description: A type of the objects to be searched for.
  - id: referencePropertyName
    type: System.String
    description: A string value representing the name of the property that can be set to the target object. This property must belong to the class specified by the _looksFor_ parameter.
  - id: messageTemplate
    type: System.String
    description: A string value representing information defining the current rule. This value is set for the [RuleBaseAttribute.CustomMessageTemplate](xref:DevExpress.Persistent.Validation.RuleBaseAttribute.CustomMessageTemplate) property.
