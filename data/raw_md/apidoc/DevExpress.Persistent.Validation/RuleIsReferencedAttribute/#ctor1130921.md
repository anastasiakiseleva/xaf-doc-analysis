  - id: targetContexts
    type: DevExpress.Persistent.Validation.DefaultContexts

    type: System.Type
    description: The type of the objects to be tested.
  - id: referencePropertyName
    type: System.String
    description: A string value representing the name of the property that can be set to the target object. This property must belong to the class specified by the _looksFor_ parameter.
