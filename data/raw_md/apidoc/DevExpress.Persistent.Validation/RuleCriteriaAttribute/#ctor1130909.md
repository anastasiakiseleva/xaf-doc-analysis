syntax:
  content: public RuleCriteriaAttribute(string id, DefaultContexts targetContexts, string criteria)
  parameters:
  - id: id
    type: System.String
    description: A string value representing the current rule identifier. You can pass an empty value as the _id_ parameter. An appropriate identifier will be generated automatically, and set for the **ID** property of the **Validation** | **Rules** | **Rule** node in the [Application Model](xref:112580).
  - id: targetContexts
    type: DevExpress.Persistent.Validation.DefaultContexts
    description: A **DefaultContexts** enumeration value that specifies a context for checking the current rule. This value is set for the [RuleBaseAttribute.TargetContextIDs](xref:DevExpress.Persistent.Validation.RuleBaseAttribute.TargetContextIDs) property.
  - id: criteria
    type: System.String
    description: A string value representing a criteria that must be satisfied by target class' objects. To learn how to set a criteria, refer to the [Ways to Build Criteria](xref:113052) topic.
