syntax:
  content: public RuleValueComparisonAttribute(string id, DefaultContexts targetContexts, ValueComparisonType operatorType, string rightOperand, string messageTemplate, ParametersMode mode)
  parameters:
  - id: id
    type: System.String
    description: A string value representing the current rule identifier. You can pass an empty value as the _id_ parameter. An appropriate identifier will begenerated automatically, and set for the **ID** property of the **Validation** | **Rules** | **Rule** node in the [Application Model](xref:112580).
  - id: targetContexts
    type: DevExpress.Persistent.Validation.DefaultContexts
    description: A **DefaultContexts** enumeration value that specifies a context for checking the current rule. This value is set for the [RuleBaseAttribute.TargetContextIDs](xref:DevExpress.Persistent.Validation.RuleBaseAttribute.TargetContextIDs) property.
  - id: operatorType
    type: DevExpress.Persistent.Validation.ValueComparisonType
    description: A [](xref:DevExpress.Persistent.Validation.ValueComparisonType) enumeration value that represents the operator type to be used when comparing values.
  - id: rightOperand
    type: System.String
    description: The string representation of an [expression](xref:4928) whose value to be compared with the target property's value.
  - id: messageTemplate
    type: System.String
    description: A string value representing information defining the current rule. This value is set for the [RuleBaseAttribute.CustomMessageTemplate](xref:DevExpress.Persistent.Validation.RuleBaseAttribute.CustomMessageTemplate) property.
  - id: mode
    type: DevExpress.Persistent.Validation.ParametersMode
    description: A [](xref:DevExpress.Persistent.Validation.ParametersMode) enumeration value specifying how the _rightOperand_ parameter value is treated.
