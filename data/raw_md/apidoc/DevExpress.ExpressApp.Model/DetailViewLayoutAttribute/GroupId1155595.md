name: GroupId
type: Property
summary: Gets the group identifier in which the editor of the target property is placed.
syntax:
  content: public string GroupId { get; }
  parameters: []
  return:
    type: System.String
    description: A string value specifying the group identifier in which the editor of the target property is placed. If a group with the specified identifier is missing in the default layout, it is created and added to the **Main** group. The type and index of the created group are specified using the [DetailViewLayoutAttribute.GroupType](xref:DevExpress.ExpressApp.Model.DetailViewLayoutAttribute.GroupType) and [DetailViewLayoutAttribute.GroupIndex](xref:DevExpress.ExpressApp.Model.DetailViewLayoutAttribute.GroupIndex) properties.
