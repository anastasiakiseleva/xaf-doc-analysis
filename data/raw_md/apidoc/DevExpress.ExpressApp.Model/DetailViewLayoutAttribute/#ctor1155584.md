  content: public DetailViewLayoutAttribute(string groupId, LayoutGroupType groupType, int groupIndex)
  parameters:
  - id: groupId
    type: System.String
    description: A string value specifying the group identifier in which the editor of the target property is placed. If a group with the specified identifier is missing in the default layout, it is created and added to the **Main** group. The type and index of the created group are specified using the _groupType_ and _groupIndex_ parameters.
  - id: groupType
    type: DevExpress.ExpressApp.Model.LayoutGroupType
    description: A [](xref:DevExpress.ExpressApp.Model.LayoutGroupType) enumeration value specifying the type of a new group which is created if a group with the  _groupId_ identifier is missing in the default layout.
  - id: groupIndex
    type: System.Int32
    description: An integer value specifying the index of a new group which is created if a group with the _groupId_ identifier is missing in the default layout.
