  content: public virtual void EnableObjectDeletionOnRemove(object collection, bool enable)
  parameters:
  - id: collection
    type: System.Object

  - id: enable

    description: '**true**, to enable the deletion from the database along with the removal from the collection; **false**, to disable it.'
