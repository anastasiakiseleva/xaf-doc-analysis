syntax:
  content: public FileTypeFilterAttribute(string filterID, string filterCaption, int index, params string[] extensions)
  parameters:
  - id: filterID
    type: System.String
    description: A string specifying the file type filter identifier. This parameter's value is assigned to the [FileTypeFilterAttribute.FilterID](xref:DevExpress.Persistent.Base.FileTypeFilterAttribute.FilterID) property.
  - id: filterCaption
    type: System.String
    description: A string specifying the file type filter description. This parameter's value is assigned to the [FileTypeFilterAttribute.FilterCaption](xref:DevExpress.Persistent.Base.FileTypeFilterAttribute.FilterCaption) property.
  - id: index
    type: System.Int32
    description: An integer value specifying the index of the file type filter. This index allows you to arrange the type filters in the required order, when multiple **FileTypeFilter** attributes are applied.
  - id: extensions
    type: System.String[]
    description: A variable number of strings that specify the type filter's file extensions. These extensions can be retrieved via the [FileTypeFilterAttribute.GetExtensions](xref:DevExpress.Persistent.Base.FileTypeFilterAttribute.GetExtensions) method.
